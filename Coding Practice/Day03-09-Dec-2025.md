## ðŸ”¹ Terraform â€” Security Group (r6i.4xlarge EC2)

provider "aws" {
  region = "us-east-1"
}

resource "aws_security_group" "PathnexSG" {
  name        = "PathnexSG"
  description = "Pathnex SG for SSH"

  ingress {             # it tells controlling traffic entering in EC2 instance except port 22.
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

resource "aws_instance" "PathnexEC2" {
  ami                    = "ami-0abcd1234abcd1234"
  instance_type          = "r6i.4xlarge"
  vpc_security_group_ids = [aws_security_group.PathnexSG.id]   # it indicates attaching security group to EC2 instances .

  tags = {
    Name = "Pathnex-EC2"
  }
}

# NodeJS Build
## ðŸ”¹ Jenkins Pipeline â€” NodeJS Build
You will learn how to **install dependencies, run tests, and build a NodeJS project**.

pipeline {
    agent any
    tools {
        nodejs 'NodeJS-16'   # tool name in ansible node js version 16.
    }
    stages {
        stage('Checkout') {
            steps {
                git url: 'https://github.com/Pathnex/sample-node-app.git'
            }
        }
        stage('Install Dependencies') {    # this stage is responsible for installing node js packages.
            steps {
                sh 'npm install'   # download dependencies listed in jason package.
            }
        }
        stage('Run Tests') {
            steps {
                sh 'npm test'
            }
        }
        stage('Build') {
            steps {
                sh 'npm run build'
            }
        }
    }
}
